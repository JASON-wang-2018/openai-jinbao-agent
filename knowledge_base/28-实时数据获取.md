## 一、实时数据概述

### 1.1 实时 vs 盘后数据

| 类型 | 延迟 | 稳定性 | 用途 |
|------|------|--------|------|
| **实时数据** | 秒级 | 一般 | 盘中操作、实时监控 |
| **盘后数据** | T+1 | 高 | 复盘分析、选股回测 |

### 1.2 数据源对比

| 数据源 | 类型 | 延迟 | 稳定性 | 免费 |
|--------|------|------|--------|------|
| 新浪财经 | 实时 | 1-3秒 | ⭐⭐⭐ | ✅ |
| 东方财富 | 实时 | 1-5秒 | ⭐⭐⭐ | ✅ |
| 同花顺 | 实时 | 1-5秒 | ⭐⭐⭐ | ✅ |
| AKShare | 实时 | 1-5秒 | ⭐⭐ | ✅ |
| Baostock | 盘后 | T+1 | ⭐⭐⭐⭐ | ✅ |
| Tushare | 实时/盘后 | 秒级/T+1 | ⭐⭐⭐⭐ | ❌ |

```python
    """批量获取多只股票"""

        symbols = symbols[:80]

                    'price': float(data[3]),

        return stocks
```

### 5.2 限流控制

```python
import time
from datetime import datetime

class RateLimiter:
    """限流器"""
    
    def __init__(self, max_calls=100, period=60):
        self.max_calls = max_calls
        self.period = period
        self.calls = []
    
    def wait(self):
        """等待可用"""
        now = time.time()
        
        # 清理过期记录
        self.calls = [t for t in self.calls if now - t < self.period]
        
        # 达到上限，等待
        if len(self.calls) >= self.max_calls:
            sleep_time = self.period - (now - self.calls[0])
            if sleep_time > 0:
                time.sleep(sleep_time)
                self.calls = self.calls[1:]
        
        self.calls.append(time.time())
    
    def get(self, func):
        """带限流的函数调用"""
        self.wait()
        return func()
```

### 5.3 断线重连

```python
class RetryHandler:
    """重试处理器"""
    
    def __init__(self, max_retries=3, delay=1):
        self.max_retries = max_retries
        self.delay = delay
    
    def execute(self, func, *args, **kwargs):
        """带重试的函数执行"""
        for i in range(self.max_retries):
            try:
                return func(*args, **kwargs)
            except Exception as e:
                if i < self.max_retries - 1:
                    time.sleep(self.delay * (i + 1))
                    print(f"第{i+1}次重试: {e}")
                else:
                    print(f"重试{self.max_retries}次后失败: {e}")
                    raise
```

## 六、实时监控

### 6.1 价格监控

```python
class PriceMonitor:
    """价格监控"""
    
    def __init__(self):
        self.thresholds = {}
        self.callbacks = []
    
    def set_alert(self, code, high=None, low=None, change=None):
        """设置预警"""
        self.thresholds[code] = {
            'high': high,
            'low': low,
            'change': change
        }
    
    def add_callback(self, callback):
        """添加回调"""
        self.callbacks.append(callback)
    
    def check(self, data):
        """检查预警"""
        code = data['code']
        if code not in self.thresholds:
            return
        
        t = self.thresholds[code]
        alerts = []
        
        if t['high'] and data['price'] >= t['high']:
            alerts.append(f"突破上限: {data['price']} >= {t['high']}")
        
        if t['low'] and data['price'] <= t['low']:
            alerts.append(f"跌破下限: {data['price']} <= {t['low']}")
        
        if t['change'] and abs(data['change']) >= t['change']:
            alerts.append(f"涨跌幅超限: {data['change']}%")
        
        for alert in alerts:
            for cb in self.callbacks:
                cb(code, alert, data)
```

```python
class MarketMonitor:
    """市场监控"""
    
    def __init__(self):
        self.data = {}
    
    def update(self, code, data):
        """更新数据"""
        self.data[code] = data
    
    def get_top_gainers(self, n=10):
        """涨幅榜"""
        sorted_stocks = sorted(
            self.data.items(),
            key=lambda x: x[1]['change'],
            reverse=True
        )
        return [data for code, data in sorted_stocks[:n]]
    
    def get_top_losers(self, n=10):
        """跌幅榜"""
        sorted_stocks = sorted(
            self.data.items(),
            key=lambda x: x[1]['change']
        )
        return [data for code, data in sorted_stocks[:n]]
    
    def get_limit_up(self):
        """涨停股"""
        return [data for data in self.data.values() if data['change'] >= 9.5]
    
    def get_limit_down(self):
        """跌停股"""
```

## 九、完整代码示例

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
实时数据获取器
多源容错 + 限流控制 + 断线重连
"""

import time
import requests
import json
from datetime import datetime
from typing import Dict, List, Optional

class RealtimeDataFetcher:
    """实时数据获取器"""
    
    def __init__(self):
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'Mozilla/5.0',
            'Referer': 'https://finance.sina.com.cn'
        })
        self.last_request = 0
        self.min_interval = 0.3  # 最小请求间隔
    
    def _rate_limit(self):
        """限流控制"""
        now = time.time()
        if now - self.last_request < self.min_interval:
            time.sleep(self.min_interval - (now - self.last_request))
        self.last_request = now
    
    def _fetch(self, url: str) -> Optional[str]:
        """带重试的请求"""
        for i in range(3):
            try:
                self._rate_limit()
                r = self.session.get(url, timeout=10)
                if r.status_code == 200:
                    return r.text
                else:
                    print(f"请求失败: {r.status_code}")
            except Exception as e:
                print(f"第{i+1}次重试: {e}")
                time.sleep(1 * (i + 1))
        return None
    
    def get_index(self, symbol: str) -> Optional[Dict]:
        """获取指数"""
        url = f"https://hq.sinajs.cn/list={symbol}"
        text = self._fetch(url)
        
        if not text:
            return None
        
        try:
            data = text.split('"')[1].split(',')
            return {
                'symbol': symbol,
                'price': float(data[3]),
                'change': (float(data[3]) - float(data[2])) / float(data[2]) * 100,
                'open': float(data[1]),
                'prev_close': float(data[2]),
                'high': float(data[4]),
                'low': float(data[5]),
                'volume': int(data[8]) * 100,
                'amount': float(data[9]) * 10000,
                'time': data[40]
            }
        except Exception as e:
            print(f"解析失败: {e}")
            return None
    
    def get_indices(self, symbols: List[str]) -> List[Dict]:
        """批量获取指数"""
        results = []
        for symbol in symbols:
            data = self.get_index(symbol)
            if data:
                results.append(data)
        return results


if __name__ == '__main__':
    fetcher = RealtimeDataFetcher()
    
    indices = ['sh000001', 'sz399001', 'sz399006']
    
    while True:
        try:
            results = fetcher.get_indices(indices)
            now = datetime.now().strftime('%H:%M:%S')
            print(f"\n【{now}】")
            for data in results:
                sign = '+' if data['change'] >= 0 else ''
                print(f"{data['symbol']}: {data['price']:.2f} ({sign}{data['change']:.2f}%)")
            time.sleep(3)
        except KeyboardInterrupt:
            print("\n已停止")
            break
        except Exception as e:
            print(f"错误: {e}")
            time.sleep(5)
```
