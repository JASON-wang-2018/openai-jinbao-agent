## 三、AKShare - 实时数据 (推荐)

### 3.1 实时行情
```python
import akshare as ak

# 大盘指数
df = ak.stock_zh_index_spot_em(symbol="sh000001")

# 个股实时
df = ak.stock_zh_a_spot_em()

# 板块行情
df = ak.stock_board_industry_name_em()
```

### 3.2 历史数据
```python
# 日线数据
df = ak.stock_zh_a_hist(
    symbol="000001",
    period="daily",
    start_date="20240101",
    end_date="20240630",
    adjust="qfq"  # 前复权
)

# 分时数据
df = ak.stock_zh_a_hist_min_em(
    symbol="000001",
    start_date="20240601",
    end_date="20240630",
    period="5"
)
```

### 3.3 资金流向
```python
# 主力资金流向
df = ak.stock_market_fund_flow_summary()

# 个股资金流向
df = ak.stock_individual_fund_flow(symbol="000001", market="sz")
```

### 3.4 涨停板数据
```python
# 涨停池
df = ak.stock_zt_pool_em(date="20240617")

# 跌停池
df = ak.stock_dt_pool_em(date="20240617")
```

## 四、BaoStock - 历史/财务 (推荐)

### 4.1 登录/登出
```python
import baostock as bs

# 登录
lg = bs.login()
print(lg.error_msg)

# 登出
bs.logout()
```

### 4.2 日线历史数据
```python
import baostock as bs

rs = bs.query_history_k_data_plus(
    "sh.600000",
    "date,code,open,high,low,close,volume,amount",
    start_date='2024-01-01',
    end_date='2024-06-30',
    frequency="d",
    adjustflag="2"  # 前复权
)

data_list = []
while (rs.error_code == '0') & rs.next():
    data_list.append(rs.get_row_data())

import pandas as pd
df = pd.DataFrame(data_list, columns=rs.fields)
```

### 4.3 财务数据
```python
# 盈利能力
rs = bs.query_profit_data(code="sh.600000", year=2023, quarter=4)

# 成长能力
rs = bs.query_growth_data(code="sh.600000", year=2023, quarter=4)

# 估值指标
rs = bs.query_valuation_data(code="sh.600000", year=2023, quarter=4)
```

### 4.4 复权因子
```python
rs = bs.query_adjust_factor(code="sh.600000", start_date='2024-01-01')
```

## 五、TuShare - 专业数据 (需要 Token)

### 5.1 获取 Token
1. 访问 https://tushare.pro 注册
2. 获取个人 token
3. 设置环境变量：`export TUSHARE_TOKEN=你的token`

### 5.2 基本使用
```python
import tushare as ts

# 设置 token
pro = ts.pro_api('your_token')

# 交易日历
df = pro.trade_cal(exchange='SSE', start_date='20240101', end_date='20240630')

# 股票列表
df = pro.stock_basic(exchange='', list_status='L', fields='ts_code,name,area,industry')

# 日线行情
df = pro.daily(ts_code='000001.SZ', start_date='20240101', end_date='20240630')
```

## 六、统一数据获取脚本

```python
"""
stock_data_manager.py 中的核心方法
"""

class StockDataManager:
    """统一股票数据管理器"""
    
    def __init__(self):
        self.source = "akshare"  # 默认
    
    def get_realtime_quote(self, code=None):
        """实时行情 - AKShare"""
        import akshare as ak
        
        if code:
            df = ak.stock_zh_a_spot_em()
            return df[df['代码'] == code]
        return ak.stock_zh_a_spot_em()
    
    def get_index_quote(self, symbol="sh000001"):
        """大盘指数 - AKShare"""
        import akshare as ak
        return ak.stock_zh_index_spot_em(symbol=symbol)
    
    def get_historical_data(self, code, start_date, end_date, adjust="qfq"):
        """历史数据 - AKShare"""
        import akshare as ak
        return ak.stock_zh_a_hist(
            symbol=code,
            period="daily",
            start_date=start_date,
            end_date=end_date,
            adjust=adjust
        )
    
    def get_financial_data(self, code, year, quarter):
        """财务数据 - BaoStock"""
        import baostock as bs
        
        bs.login()
        try:
            rs = bs.query_profit_data(code=code, year=year, quarter=quarter)
            data_list = []
            while (rs.error_code == '0') & rs.next():
                data_list.append(rs.get_row_data())
            import pandas as pd
            return pd.DataFrame(data_list, columns=rs.fields)
        finally:
            bs.logout()
    
    def get_market_sentiment(self, date=None):
        """市场情绪 - AKShare"""
        import akshare as ak
        from datetime import datetime
        
        date = date or datetime.now().strftime('%Y-%m-%d')
        
        try:
            df = ak.stock_zt_pool_em(date=date)
            return {
                'zt_count': len(df),
                'zt_list': df['代码'].tolist() if not df.empty else []
            }
        except:
            return {'zt_count': 0, 'zt_list': []}
    
    def get_sector_performance(self):
        """板块表现 - AKShare"""
        import akshare as ak
        return ak.stock_board_industry_name_em()
```
