## 一、数据库选择

### 1.1 推荐方案

| 用途 | 推荐数据库 | 原因 |
|------|-----------|------|
| 历史数据 | MySQL/PostgreSQL | 稳定、成熟 |
| 实时数据 | InfluxDB | 时序数据优化 |
| 缓存数据 | Redis | 高速读写 |
| 文件数据 | SQLite | 轻量、免配置 |
| 向量数据 | ChromaDB | 知识库检索 |

### 1.2 轻量方案（推荐）

```
数据量 < 100万条 → SQLite + 文件
数据量 > 100万条 → MySQL
```

### 2.3 Python操作

```python
import pymysql

class MySQLDB:
    def __init__(self):
        self.conn = pymysql.connect(
            host='localhost',
            user='root',
            password='123456',
            database='stock_db',
            charset='utf8mb4'
        )
    
    def insert_daily(self, data):
        """插入日线数据"""
        sql = """
        INSERT INTO daily_data 
        (code, trade_date, open, high, low, close, volume, amount)
        VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
        ON DUPLICATE KEY UPDATE 
        open=VALUES(open), high=VALUES(high), low=VALUES(low),
        close=VALUES(close), volume=VALUES(volume), amount=VALUES(amount)
        """
        with self.conn.cursor() as cursor:
            cursor.executemany(sql, data)
        self.conn.commit()
    
    def query_daily(self, code, start_date, end_date):
        """查询日线数据"""
        sql = """
        SELECT * FROM daily_data 
        WHERE code=%s AND trade_date BETWEEN %s AND %s
        ORDER BY trade_date
        """
        with self.conn.cursor() as cursor:
            cursor.execute(sql, (code, start_date, end_date))
            return cursor.fetchall()
    
    def close(self):
        self.conn.close()
```

## 三、SQLite方案

### 3.1 创建数据库

```python
import sqlite3

class SQLiteDB:
    def __init__(self, db_file='stock.db'):
        self.conn = sqlite3.connect(db_file)
        self.create_tables()
    
    def create_tables(self):
        """创建表"""
        sql = """
        CREATE TABLE IF NOT EXISTS daily_data (
            code TEXT,
            trade_date TEXT,
            open REAL,
            high REAL,
            low REAL,
            close REAL,
            volume INTEGER,
            amount REAL,
            PRIMARY KEY (code, trade_date)
        );
        
        CREATE TABLE IF NOT EXISTS stocks (
            code TEXT PRIMARY KEY,
            name TEXT,
            industry TEXT
        );
        """
        self.conn.executescript(sql)
        self.conn.commit()
    
    def insert_daily(self, data_list):
        """批量插入"""
        sql = """
        INSERT OR REPLACE INTO daily_data 
        (code, trade_date, open, high, low, close, volume, amount)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        """
        self.conn.executemany(sql, data_list)
        self.conn.commit()
    
    def query_daily(self, code, start_date, end_date):
        """查询"""
        sql = """
        SELECT * FROM daily_data 
        WHERE code=? AND trade_date BETWEEN ? AND ?
        ORDER BY trade_date
        """
        return self.conn.execute(sql, (code, start_date, end_date)).fetchall()
```

## 四、Redis缓存

### 4.1 安装与使用

```python
import redis

class RedisCache:
    def __init__(self):
        self.r = redis.Redis(host='localhost', port=6379, db=0)
    
    def set_price(self, code, price):
        """缓存价格"""
        self.r.setex(f"price:{code}", 60, price)  # 60秒过期
    
    def get_price(self, code):
        return self.r.get(f"price:{code}")
    
    def set_daily(self, code, date, data):
        """缓存日线"""
        self.r.hset(f"daily:{code}", date, str(data))
    
    def get_daily(self, code, date):
        return self.r.hget(f"daily:{code}", date)
```

| 龙虎榜 | 每日 | 文件 |

### 5.2 自动更新脚本

```python
"""
每日数据更新脚本
cron: 16:00 执行
"""

import akshare as ak
import pandas as pd
from datetime import datetime

class DailyUpdater:
    def __init__(self, db):
        self.db = db
    
    def update_all(self):
        """更新所有股票"""
        # 获取股票列表
        stocks = ak.stock_zh_a_spot_em()['代码'].tolist()
        
        for code in stocks[:100]:  # 先更新100只
            try:
                self.update_stock(code)
                print(f"{code} 更新完成")
            except Exception as e:
                print(f"{code} 更新失败: {e}")
    
    def update_stock(self, code):
        """更新单只股票"""
        today = datetime.now().strftime('%Y%m%d')
        
        # 获取日线
        df = ak.stock_zh_a_hist(
            symbol=code,
            period="daily",
            start_date=today,
            end_date=today,
            adjust="qfq"
        )
        
        if len(df) > 0:
            data = [
                (code, row['日期'], row['开盘'], row['最高'], row['最低'],
                 row['收盘'], row['成交量'], row['成交额'])
                for _, row in df.iterrows()
            ]
            self.db.insert_daily(data)
```

## 六、数据备份

### 6.1 备份策略

```bash
#!/bin/bash
# backup.sh - 每日备份

DATE=$(date +%Y%m%d)
BACKUP_DIR=/backup/stock

# 备份MySQL
mysqldump -u root -p123456 stock_db > $BACKUP_DIR/stock_$DATE.sql

# 压缩
gzip $BACKUP_DIR/stock_$DATE.sql

# 保留30天
find $BACKUP_DIR -name "*.sql.gz" -mtime +30 -delete
```

## 十、常见问题

|------|---------|
